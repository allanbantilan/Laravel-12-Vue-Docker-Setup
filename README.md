# -----------------------------
# Helpful Notes
# -----------------------------
# Usage Examples:
#   make up                # Start containers
#   make down              # Stop containers
#   make restart           # Restart containers
#   make migrate           # Run migrations
#   make migrate-refresh   # Refresh database
#   make tinker            # Open Tinker
#   make shell             # Enter container shell
#   make artisan cmd="route:list"   # Run Artisan command
#   make composer cmd="install"     # Run Composer command
#   make npm cmd="run dev"          # Run npm command
#   make dev                # Start Vue dev server
#   make build              # Build Vue assets
#   make test               # Run tests
